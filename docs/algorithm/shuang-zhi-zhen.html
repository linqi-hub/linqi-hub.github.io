    # 剑指offer-22-链表中倒数第k个节点

    # 思路

    • 先设置一个快指针，先行k步，然后慢指针在从头遍历；
    • 此时快慢指针同时往后走；
    • 当快指针为空时，就是倒数第k个节点；

    # 上代码

    var getKthFromEnd = function (head, k) {
  let p1 = head
  let p2 = head
  while (k > 0) {
    p1 = p1.next
    k--
  }
  while (p1) {
    p1 = p1.next
    p2 = p2.next
  }
  return p2
}

    # 剑指offer-52-两个链表的第一个公共节点

    # 思路

    • 两个链表的公共节点，只有一种情况，就是两个链表相交后，就成为一条链表(不会有其他情况)；
    • 首先找出两个链表的差值，注意遍历的时候要使用新建的指针，保存原来的头结点；
    • 然后确定好两个链表对应的指针，让快的先走差值，此时就已经同步；
    • 两链表同时走，当节点相同的时候，直接返回，否则继续遍历；

    # 上代码

    var getIntersectionNode = function (headA, headB) {
  let n = 0
  let cur1 = headA
  let cur2 = headB
  while (cur1) {
    cur1 = cur1.next
    n++
  }
  while (cur2) {
    cur2 = cur2.next
    n--
  }
  if (cur1 != cur2) return null
  cur1 = n > 0 ? headA : headB
  cur2 = cur1 == headA ? headB : headA
  n = Math.abs(n)
  while (n > 0) {
    cur1 = cur1.next
    n--
  }
  while (cur1 != cur2) {
    cur1 = cur1.next
    cur2 = cur2.next
  }
  return cur1
}

    # 剑指offer-57-和为s的两个数字

    # 思路

    • 首先是递增排序的列表，并且是查找，所以用双指针；
    • 首尾共同遍历，若两数之和大于目标值，则尾部往前移动，若两数之和小于目标值，则头部往后移动；

    # 上代码

    var twoSum = function (nums, target) {
  let i = 0
  let j = nums.length - 1
  while (i < j) {
    let sum = nums[i] + nums[j]
    if (sum == target) {
      return [nums[i], nums[j]]
    } else if (sum > target) {
      j--
    } else {
      i++
    }
  }
  return null
}
